// Máscara do CPF
function maskCPF(value: string) {
  const digits = (value || "").replace(/\D/g, "").slice(0, 11)
  const p1 = digits.slice(0, 3)
  const p2 = digits.slice(3, 6)
  const p3 = digits.slice(6, 9)
  const p4 = digits.slice(9, 11)

  let out = p1
  if (p2) out += "." + p2
  if (p3) out += "." + p3
  if (p4) out += "-" + p4
  return out
}

//Máscara do telefone
function maskPhoneBR(value: string) {
  const digits = (value || "").replace(/\D/g, "").slice(0, 11) // DDD + 9 dígitos
  const ddd = digits.slice(0, 2)
  const part1 = digits.slice(2, 7)  // 5 dígitos (celular)
  const part2 = digits.slice(7, 11) // 4 dígitos

  let out = ""
  if (ddd) out = `(${ddd}`
  if (digits.length >= 3) out += ") "
  if (part1) out += part1
  if (part2) out += "-" + part2

  return out
}

// ✅ Regras de obrigatoriedade de anexos por módulo (descrição padronizada)
const REQUIRED_DOCS_BY_MODULE: Record<string, string[]> = {
  hemodialise: [
    "RG",
    "CNS",
    "EXAME DE HIV",
    "HEPATITE B",
    "HEPATITE C",
    "EXAMES MENSAIS (ROTINA DO SERVIÇO)",
    "COMPROVANTE DE RESIDÊNCIA",
  ],
  tfd: [
    "LAUDO MÉDICO PARA SOLICITAÇÃO DE TFD (MODELO)",
    "RG",
    "CNS",
  ],
}

// Normaliza texto pra comparar (ignora acento/caixa/espaço)
function norm(s: string) {
  return (s || "")
    .normalize("NFD")
    .replace(/[\u0300-\u036f]/g, "")
    .toUpperCase()
    .replace(/\s+/g, " ")
    .trim()
}

function getMissingRequiredDocs(params: {
  modulo: string
  arquivos: { descricao: string }[]
  comprovanteNaoNomeSolicitante?: boolean
  tfdDeclaracaoNaoManuscrita?: boolean
}) {
  const { modulo, arquivos, comprovanteNaoNomeSolicitante, tfdDeclaracaoNaoManuscrita } = params

  const required = [...(REQUIRED_DOCS_BY_MODULE[modulo] || [])]

  // Regra extra: Hemodiálise -> se comprovante não estiver no nome, exigir carta
  if (modulo === "hemodialise" && comprovanteNaoNomeSolicitante) {
    required.push("CARTA DE PRÓPRIO PUNHO")
  }

  // Regra extra: TFD -> não pode ser manuscrita (vamos checar a declaração)
  // Aqui só devolvemos um “faltando” simbólico pra você mostrar bonito no toast
  if (modulo === "tfd" && !tfdDeclaracaoNaoManuscrita) {
    required.push("DECLARAÇÃO: LAUDO NÃO MANUSCRITO")
  }

  const existing = new Set(arquivos.map(a => norm(a.descricao)))

  const missing = required.filter(r => !existing.has(norm(r)))

  return missing
}

              <CardContent className="flex flex-col gap-6">
                {/* Instituicao */}
                <div>
                  <h3 className="font-semibold text-foreground mb-3 flex items-center gap-2">
                    <Building2 className="h-4 w-4" /> Instituicao Solicitante
                  </h3>
                  <div className="grid gap-4 md:grid-cols-2">
                    <div>
                      <Label>Nome da Unidade *</Label>
                      <Input {...register("localSolicitante")} />
                      {errors.localSolicitante && <p className="text-sm text-destructive mt-1">{errors.localSolicitante.message}</p>}
                    </div>
                    <div>
                      <Label>E-mail da Instituicao *</Label>
                      <Input type="email" {...register("emailSolicitante")} />
                      {errors.emailSolicitante && <p className="text-sm text-destructive mt-1">{errors.emailSolicitante.message}</p>}
                    </div>
                  </div>
                  <div className="mt-2">
                    <Label>Telefone(s) da Instituicao *</Label>
                    <div className="flex flex-col gap-2 mt-1">
                      {telSolFields.map((field, i) => (
                        <div key={field.id} className="flex gap-2">
                          <Input placeholder="(00) 0000-0000" {...register(`telefoneSolicitante.${i}.value`)} />
                          {telSolFields.length > 1 && (
                            <Button type="button" variant="ghost" size="icon" onClick={() => removeTelSol(i)}>
                              <Trash2 className="h-4 w-4" />
                            </Button>
                          )}
                        </div>
                      ))}
                      <Button type="button" variant="outline" size="sm" onClick={() => appendTelSol({ value: "" })} className="w-fit bg-transparent">
                        <Plus className="mr-1 h-3 w-3" /> Adicionar telefone
                      </Button>
                    </div>
                  </div>
                </div>

                <Separator />

                {/* Modulo e tipo */}
                <div className="grid gap-4 md:grid-cols-2">
                  <div>
                    <Label>Modulo *</Label>
                    <Select value={watch("modulo")} onValueChange={(v) => setValue("modulo", v, { shouldDirty: true, shouldTouch: true, shouldValidate: true })}>
                      <SelectTrigger><SelectValue placeholder="Selecione" /></SelectTrigger>
                      <SelectContent>
                        {MODULES.map((m) => (
                          <SelectItem key={m} value={m}>{MODULE_LABELS[m]}</SelectItem>
                        ))}
                      </SelectContent>
                    </Select>
                    {errors.modulo && <p className="text-sm text-destructive mt-1">{errors.modulo.message}</p>}
                  </div>
                 
                    {watch("modulo") === "hemodialise" && (
                      <div>
                        <Label>Tipo de Solicitacao *</Label>
                        <Select
                          value={watch("tipoSolicitacao")}
                          onValueChange={(v) =>
                            setValue("tipoSolicitacao", v, { shouldDirty: true, shouldTouch: true, shouldValidate: true })
                          }
                        >
                          <SelectTrigger><SelectValue placeholder="Selecione" /></SelectTrigger>
                          <SelectContent>
                            {TIPO_SOLICITACAO.map((t) => (
                              <SelectItem key={t} value={t}>{TIPO_SOLICITACAO_LABELS[t]}</SelectItem>
                            ))}
                          </SelectContent>
                        </Select>
                        {errors.tipoSolicitacao && <p className="text-sm text-destructive mt-1">{errors.tipoSolicitacao.message}</p>}
                      </div>
                    )}




                </div>

                {/* Clinico */}
                <div className="grid gap-4 md:grid-cols-2">
                  <div>
                    <Label>Codigo SIGTAP *</Label>
                    <Input {...register("codigoSigtap")} />
                    {errors.codigoSigtap && <p className="text-sm text-destructive mt-1">{errors.codigoSigtap.message}</p>}
                  </div>
                  <div>
                    <Label>Descricao do Procedimento *</Label>
                    <Input {...register("descricaoSigtap")} />
                    {errors.descricaoSigtap && <p className="text-sm text-destructive mt-1">{errors.descricaoSigtap.message}</p>}
                  </div>
                  <div>
                    <Label>CID-10 *</Label>
                    <Input {...register("cid10")} />
                    {errors.cid10 && <p className="text-sm text-destructive mt-1">{errors.cid10.message}</p>}
                  </div>
                  <div>
                    <Label>Especialidade *</Label>
                    <Input {...register("especialidade")} />
                    {errors.especialidade && <p className="text-sm text-destructive mt-1">{errors.especialidade.message}</p>}
                  </div>
                  <div>
                    <Label>Subespecialidade</Label>
                    <Input {...register("subespecialidade")} />
                  </div>
                  <div>
                    <Label>Local Solicitado</Label>
                    <Input placeholder="Se houver preferencia" {...register("localSolicitado")} />
                  </div>
                </div>

                <div className="grid gap-4 md:grid-cols-3">
                  <div>
                    <Label>Peso (kg)</Label>
                    <Input {...register("peso")} />
                  </div>
                  <div>
                    <Label>Altura (m)</Label>
                    <Input {...register("altura")} />
                  </div>
                  <div>
                    <Label>Tipo Sanguineo</Label>
                    <Input placeholder="Ex: O+" {...register("tipoSanguineo")} />
                  </div>
                </div>

                <div>
                  <Label>Observacoes da Unidade</Label>
                  <Textarea rows={3} {...register("observacoesUnidade")} />
                </div>

                <div className="flex items-center gap-2">
                  <Checkbox
                    id="acaoJudicial"
                    checked={watch("acaoJudicial") || false}
                    onCheckedChange={(v) => setValue("acaoJudicial", v === true)}
                  />
                  <Label htmlFor="acaoJudicial" className="cursor-pointer">Acao Judicial?</Label>
                </div>

                <div className="flex justify-between">
                  <Button type="button" variant="outline" onClick={() => setStep(1)} className="bg-transparent">
                    <ArrowLeft className="mr-1 h-4 w-4" /> Voltar
                  </Button>
                  <Button type="button" onClick={nextStep}>Proximo: Documentos</Button>
                </div>
              </CardContent>
            </Card>
          )}

          {/* Step 3: Documentos */}
          {step === 3 && (
            <Card>
              <CardHeader>
                <CardTitle className="flex items-center gap-2 text-foreground">
                  <Paperclip className="h-5 w-5" />
                  Documentos
                </CardTitle>
                <CardDescription>Anexe os documentos necessarios (pessoais, laudos, checklist, outros)</CardDescription>
              </CardHeader>
              <CardContent className="flex flex-col gap-4">

                {/* ✅ Aviso + Lista de documentos obrigatórios (com *) */}
                {moduloSelecionado && (
                  <div className="rounded-lg border p-4 bg-muted/30">
                    <div className="flex items-center justify-between gap-3">
                      <div>
                        <p className="font-semibold text-foreground">
                          Documentos obrigatórios para {MODULE_LABELS[moduloSelecionado as Module]}
                        </p>
                        <p className="text-sm text-muted-foreground">
                          Os itens com <b>*</b> são obrigatórios. Clique em <b>Anexar</b> para preencher automaticamente a descrição padronizada.
                        </p>
                      </div>
                      {requiredMissing.length === 0 ? (
                        <Badge variant="secondary" className="bg-emerald-100 text-emerald-800">
                          Tudo OK
                        </Badge>
                      ) : (
                        <Badge variant="secondary" className="bg-amber-100 text-amber-800">
                          Faltam {requiredMissing.length}
                        </Badge>
                      )}
                    </div>

                    {/* Regras extras */}
                    {moduloSelecionado === "hemodialise" && (
                      <div className="mt-3 flex items-center gap-2 text-sm">
                        <input
                          type="checkbox"
                          checked={compResidNaoNomeSolicitante}
                          onChange={(e) => setCompResidNaoNomeSolicitante(e.target.checked)}
                        />
                        <span>
                          Comprovante de residência <b>não</b> está no nome do solicitante
                          <span className="ml-1 text-muted-foreground">(vai exigir “Carta de próprio punho”)</span>
                        </span>
                      </div>
                    )}

                    {moduloSelecionado === "tfd" && (
                      <div className="mt-3 text-sm">
                        <div className="flex items-center gap-2">
                          <input
                            type="checkbox"
                            checked={tfdDeclaracaoNaoManuscrita}
                            onChange={(e) => setTfdDeclaracaoNaoManuscrita(e.target.checked)}
                          />
                          <span>
                            Declaro que o <b>laudo de TFD não é manuscrito</b>.
                          </span>
                        </div>
                        <div className="mt-2 text-sm text-muted-foreground">
                          Modelo para download:{" "}
                          <a className="underline" href="/modelos/laudo-tfd.pdf" target="_blank" rel="noreferrer">
                            Laudo TFD (PDF)
                          </a>
                        </div>
                      </div>
                    )}

                    <ul className="mt-4 space-y-2">
                      {requiredDocs.map((doc) => {
                        const ok = arquivos.some((a) => norm(a.descricao) === norm(doc))
                        return (
                          <li key={doc} className="flex items-center justify-between gap-3 rounded border bg-background p-2">
                            <div className="text-sm">
                              <span className="font-semibold">{doc} *</span>
                              <span className={`ml-2 text-xs ${ok ? "text-emerald-600" : "text-amber-600"}`}>
                                {ok ? "(anexado)" : "(pendente)"}
                              </span>
                            </div>

                            {!ok ? (
                              <Button
                                type="button"
                                size="sm"
                                variant="outline"
                                className="bg-transparent"
                                onClick={() => {
                                  // padroniza categoria e descrição para bater na validação
                                  setNovoArqCategoria("pessoal")
                                  setNovoArqDescricao(doc)
                                }}
                              >
                                Anexar
                              </Button>
                            ) : (
                              <Button
                                type="button"
                                size="sm"
                                variant="ghost"
                                onClick={() => {}}
                              >
                                OK
                              </Button>
                            )}
                          </li>
                        )
                      })}
                    </ul>

                    {requiredMissing.length > 0 && (
                      <div className="mt-3 text-sm text-destructive">
                        <b>Não será possível enviar</b> enquanto os obrigatórios não forem anexados.
                      </div>
                    )}
                  </div>
                )}

                {/* Existing attachments */}
                {arquivos.length > 0 && (
                  <div className="flex flex-col gap-2">
                    {arquivos.map((arq, i) => (
                      <div key={i} className="flex items-center justify-between rounded-lg border p-3">
                        <div className="flex items-center gap-3">
                          <FileText className="h-4 w-4 text-muted-foreground" />
                          <div>
                            <p className="text-sm font-medium text-foreground">{arq.descricao}</p>
                            <Badge variant="secondary" className="text-xs">{CATEGORIA_ANEXO_LABELS[arq.categoria]}</Badge>
                          </div>
                        </div>
                        <Button type="button" variant="ghost" size="icon" onClick={() => setArquivos((p) => p.filter((_, j) => j !== i))}>
                          <Trash2 className="h-4 w-4" />
                        </Button>
                      </div>
                    ))}
                  </div>
                )}

                {/* Add new */}
                <div className="rounded-lg border border-dashed p-4 flex flex-col gap-3">
                  <p className="text-sm font-medium text-foreground">Adicionar Documento</p>
                  <div className="grid gap-3 md:grid-cols-2">
                    <div>
                      <Label>Categoria</Label>
                      <Select value={novoArqCategoria} onValueChange={(v) => setNovoArqCategoria(v as CategoriaAnexo)}>
                        <SelectTrigger><SelectValue /></SelectTrigger>
                        <SelectContent>
                          {CATEGORIA_ANEXO.map((c) => (
                            <SelectItem key={c} value={c}>{CATEGORIA_ANEXO_LABELS[c]}</SelectItem>
                          ))}
                        </SelectContent>
                      </Select>
                    </div>
                    <div>
                      <Label>Descricao do documento</Label>
                      <Input value={novoArqDescricao} onChange={(e) => setNovoArqDescricao(e.target.value)} placeholder="Ex: Laudo medico cardiologico" />
                    </div>
                  </div>
                  <div>
                    <Label className="text-muted-foreground text-xs">Arquivo (simulado para demonstracao)</Label>
                    <Input type="file" className="mt-1" />
                  </div>
                  <Button type="button" variant="outline" size="sm" onClick={addArquivo} className="w-fit bg-transparent">
                    <Plus className="mr-1 h-3 w-3" /> Adicionar documento
                  </Button>
                </div>

                <Separator />

                <div className="flex justify-between">
                  <Button type="button" variant="outline" onClick={() => setStep(2)} className="bg-transparent">
                    <ArrowLeft className="mr-1 h-4 w-4" /> Voltar
                  </Button>
                  <Button type="submit">Enviar Solicitacao</Button>
                </div>
              </CardContent>
            </Card>
          )}
        </form>

        <div className="mt-8 text-center">
          <Link href="/consulta" className="text-sm text-primary hover:underline">
            Ja tem um protocolo? Consulte aqui
          </Link>
          <span className="mx-3 text-muted-foreground">|</span>
          <Link href="/login" className="text-sm text-primary hover:underline">
            Acesso restrito (login)
          </Link>
        </div>
      </div>
    </div>
  )
}
